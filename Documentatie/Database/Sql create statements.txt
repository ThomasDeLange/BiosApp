CREATE TABLE `Buyer` ( `BuyerID` INTEGER NOT NULL, `FirstName` TEXT NOT NULL, FOREIGN KEY(`BuyerID`) REFERENCES `Ticket`(`TicketID`) )

CREATE TABLE `Seat` ( `SeatID` INTEGER NOT NULL, `RowNumber` INTEGER NOT NULL, `SeatNumber` INTEGER NOT NULL, `Field4` INTEGER, PRIMARY KEY(`SeatID`) )

CREATE TABLE "Ticket" ( `TicketID` INTEGER NOT NULL, `BuyerID` INTEGER NOT NULL, `QRCode` TEXT NOT NULL, `FilmTitle` TEXT NOT NULL, `RunTime` TEXT NOT NULL, `SeatID` INTEGER NOT NULL, FOREIGN KEY(`SeatID`) REFERENCES `Seat`(`SeatID`) ON UPDATE CASCADE ON DELETE CASCADE, PRIMARY KEY(`TicketID`), FOREIGN KEY(`BuyerID`) REFERENCES `Buyer`(`BuyerID`) ON UPDATE CASCADE ON DELETE CASCADE )
